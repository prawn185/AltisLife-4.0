private["_message"];
disableSerialization;


waitUntil {!isnull (findDisplay 9000)};


if (life_cash <= 6000 ) exitWith { systemChat "You need 60000€ for writing a message"; };
if (playerSide != civilian ) exitWith { systemChat "You have to be a civilian"; };
life_cash = life_cash - 60000;


_message = ctrlText 9001;


[[3,format ["%1 sends the following message to all: %2",name player,_message]],"life_fnc_broadcast",true,false] call life_fnc_MP;
life_channel_send = false;


[] spawn
{
    sleep 600;
    life_channel_send = true;
};
